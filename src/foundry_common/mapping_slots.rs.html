1use alloy_primitives::{B256, map::B256HashMap};
2
3/// Recorded mapping slots.
4#[derive(Clone, Debug, Default)]
5pub struct MappingSlots {
6    /// Holds mapping parent (slots => slots)
7    pub parent_slots: B256HashMap<B256>,
8
9    /// Holds mapping key (slots => key)
10    pub keys: B256HashMap<B256>,
11
12    /// Holds mapping child (slots => slots[])
13    pub children: B256HashMap<Vec<B256>>,
14
15    /// Holds the last sha3 result `sha3_result => (data_low, data_high)`, this would only record
16    /// when sha3 is called with `size == 0x40`, and the lower 256 bits would be stored in
17    /// `data_low`, higher 256 bits in `data_high`.
18    /// This is needed for mapping_key detect if the slot is for some mapping and record that.
19    pub seen_sha3: B256HashMap<(B256, B256)>,
20}
21
22impl MappingSlots {
23    /// Tries to insert a mapping slot. Returns true if it was inserted.
24    pub fn insert(&mut self, slot: B256) -> bool {
25        match self.seen_sha3.get(&slot).copied() {
26            Some((key, parent)) => {
27                if self.keys.insert(slot, key).is_some() {
28                    return false;
29                }
30                self.parent_slots.insert(slot, parent);
31                self.children.entry(parent).or_default().push(slot);
32                self.insert(parent);
33                true
34            }
35            None => false,
36        }
37    }
38}
