19/// Information about a storage slot including its label, type, and decoded values.
20#[derive(Serialize, Debug)]
21pub struct SlotInfo {
22    /// The variable name from the storage layout.
23    ///
24    /// For top-level variables: just the variable name (e.g., "myVariable")
25    /// For struct members: dotted path (e.g., "myStruct.memberName")
26    /// For array elements: name with indices (e.g., "myArray\[0\]", "matrix\[1\]\[2\]")
27    /// For nested structures: full path (e.g., "outer.inner.field")
28    /// For mappings: base name with keys (e.g., "balances\[0x1234...\]")/ex
29    pub label: String,
30    /// The Solidity type information
31    #[serde(rename = "type", serialize_with = "serialize_slot_type")]
32    pub slot_type: StorageTypeInfo,
33    /// Offset within the storage slot (for packed storage)
34    pub offset: i64,
35    /// The storage slot number as a string
36    pub slot: String,
37    /// For struct members, contains nested SlotInfo for each member
38    ///
39    /// This is populated when a struct's members / fields are packed in a single slot.
40    #[serde(skip_serializing_if = "Option::is_none")]
41    pub members: Option<Vec<SlotInfo>>,
42    /// Decoded values (if available) - used for struct members
43    #[serde(skip_serializing_if = "Option::is_none")]
44    pub decoded: Option<DecodedSlotValues>,
45    /// Decoded mapping keys (serialized as "key" for single, "keys" for multiple)
46    #[serde(
47        skip_serializing_if = "Option::is_none",
48        flatten,
49        serialize_with = "serialize_mapping_keys"
50    )]
51    pub keys: Option<Vec<String>>,
52}

54/// Wrapper type that holds both the original type label and the parsed DynSolType.
55///
56/// We need both because:
57/// - `label`: Used for serialization to ensure output matches user expectations
58/// - `dyn_sol_type`: The parsed type used for actual value decoding
59#[derive(Debug)]
60pub struct StorageTypeInfo {
61    /// The original type label from storage layout (e.g., "uint256", "address", "mapping(address
62    /// => uint256)")
63    pub label: String,
64    /// The parsed dynamic Solidity type used for decoding
65    pub dyn_sol_type: DynSolType,
66}
67
68impl SlotInfo {
69    /// Decodes a single storage value based on the slot's type information.
70    pub fn decode(&self, value: B256) -> Option<DynSolValue> {
71        // Storage values are always 32 bytes, stored as a single word
72        let mut actual_type = &self.slot_type.dyn_sol_type;
73        // Unwrap nested arrays to get to the base element type.
74        while let DynSolType::FixedArray(elem_type, _) = actual_type {
75            actual_type = elem_type.as_ref();
76        }
77
78        // Decode based on the actual type
79        actual_type.abi_decode(&value.0).ok()
80    }
81
82    /// Decodes storage values (previous and new) and populates the decoded field.
83    /// For structs with members, it decodes each member individually.
84    pub fn decode_values(&mut self, previous_value: B256, new_value: B256) {
85        // If this is a struct with members, decode each member individually
86        if let Some(members) = &mut self.members {
87            for member in members.iter_mut() {
88                let offset = member.offset as usize;
89                let size = match &member.slot_type.dyn_sol_type {
90                    DynSolType::Uint(bits) | DynSolType::Int(bits) => bits / 8,
91                    DynSolType::Address => 20,
92                    DynSolType::Bool => 1,
93                    DynSolType::FixedBytes(size) => *size,
94                    _ => 32, // Default to full word
95                };
96
97                // Extract and decode member values
98                let mut prev_bytes = [0u8; 32];
99                let mut new_bytes = [0u8; 32];
100
101                if offset + size <= 32 {
102                    // In Solidity storage, values are right-aligned
103                    // For offset 0, we want the rightmost bytes
104                    // For offset 16 (for a uint128), we want bytes 0-16
105                    // For packed storage: offset 0 is at the rightmost position
106                    // offset 0, size 16 -> read bytes 16-32 (rightmost)
107                    // offset 16, size 16 -> read bytes 0-16 (leftmost)
108                    let byte_start = 32 - offset - size;
109                    prev_bytes[32 - size..]
110                        .copy_from_slice(&previous_value.0[byte_start..byte_start + size]);
111                    new_bytes[32 - size..]
112                        .copy_from_slice(&new_value.0[byte_start..byte_start + size]);
113                }
114
115                // Decode the member values
116                if let (Ok(prev_val), Ok(new_val)) = (
117                    member.slot_type.dyn_sol_type.abi_decode(&prev_bytes),
118                    member.slot_type.dyn_sol_type.abi_decode(&new_bytes),
119                ) {
120                    member.decoded =
121                        Some(DecodedSlotValues { previous_value: prev_val, new_value: new_val });
122                }
123            }
124            // For structs with members, we don't need a top-level decoded value
125        } else {
126            // For non-struct types, decode directly
127            if let (Some(prev), Some(new)) = (self.decode(previous_value), self.decode(new_value)) {
128                self.decoded = Some(DecodedSlotValues { previous_value: prev, new_value: new });
129            }
130        }
131    }
132}

134/// Custom serializer for StorageTypeInfo that only outputs the label
135fn serialize_slot_type<S>(info: &StorageTypeInfo, serializer: S) -> Result<S::Ok, S::Error>
136where
137    S: serde::Serializer,
138{
139    serializer.serialize_str(&info.label)
140}
141
142/// Custom serializer for mapping keys
143fn serialize_mapping_keys<S>(keys: &Option<Vec<String>>, serializer: S) -> Result<S::Ok, S::Error>
144where
145    S: serde::Serializer,
146{
147    use serde::ser::SerializeMap;
148
149    if let Some(keys) = keys {
150        let mut map = serializer.serialize_map(Some(1))?;
151        if keys.len() == 1 {
152            map.serialize_entry("key", &keys[0])?;
153        } else if keys.len() > 1 {
154            map.serialize_entry("keys", keys)?;
155        }
156        map.end()
157    } else {
158        serializer.serialize_none()
159    }
160}

162/// Decoded storage slot values
163#[derive(Debug)]
164pub struct DecodedSlotValues {
165    /// Initial decoded storage value
166    pub previous_value: DynSolValue,
167    /// Current decoded storage value
168    pub new_value: DynSolValue,
169}
170
171impl Serialize for DecodedSlotValues {
172    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
173    where
174        S: serde::Serializer,
175    {
176        use serde::ser::SerializeStruct;
177
178        let mut state = serializer.serialize_struct("DecodedSlotValues", 2)?;
179        state.serialize_field("previousValue", &format_token_raw(&self.previous_value))?;
180        state.serialize_field("newValue", &format_token_raw(&self.new_value))?;
181        state.end()
182    }
183}

196    /// Identifies a storage slots type using the [`StorageLayout`].
197    ///
198    /// It can also identify whether a slot belongs to a mapping if provided with [`MappingSlots`].
199    pub fn identify(&self, slot: &B256, mapping_slots: Option<&MappingSlots>) -> Option<SlotInfo> {
200        let slot_u256 = U256::from_be_bytes(slot.0);
201        let slot_str = slot_u256.to_string();
202
203        for storage in &self.storage_layout.storage {
204            let storage_type = self.storage_layout.types.get(&storage.storage_type)?;
205            let dyn_type = DynSolType::parse(&storage_type.label).ok();
206
207            // Check if we're able to match on a slot from the layout i.e any of the base slots.
208            // This will always be the case for primitive types that fit in a single slot.
209            if storage.slot == slot_str
210                && let Some(parsed_type) = dyn_type
211            {
212                // Successfully parsed - handle arrays or simple types
213                let label = if let DynSolType::FixedArray(_, _) = &parsed_type {
214                    format!("{}{}", storage.label, get_array_base_indices(&parsed_type))
215                } else {
216                    storage.label.clone()
217                };
218
219                return Some(SlotInfo {
220                    label,
221                    slot_type: StorageTypeInfo {
222                        label: storage_type.label.clone(),
223                        dyn_sol_type: parsed_type,
224                    },
225                    offset: storage.offset,
226                    slot: storage.slot.clone(),
227                    members: None,
228                    decoded: None,
229                    keys: None,
230                });
231            }
232
233            // Encoding types: <https://docs.soliditylang.org/en/latest/internals/layout_in_storage.html#json-output>
234            if storage_type.encoding == ENCODING_INPLACE {
235                // Can be of type FixedArrays or Structs
236                // Handles the case where the accessed `slot` is maybe different from the base slot.
237                let array_start_slot = U256::from_str(&storage.slot).ok()?;
238
239                if let Some(parsed_type) = dyn_type
240                    && let DynSolType::FixedArray(_, _) = parsed_type
241                    && let Some(slot_info) = self.handle_array_slot(
242                        storage,
243                        storage_type,
244                        slot_u256,
245                        array_start_slot,
246                        &slot_str,
247                    )
248                {
249                    return Some(slot_info);
250                }
251
252                // If type parsing fails and the label is a struct
253                if is_struct(&storage_type.label) {
254                    let struct_start_slot = U256::from_str(&storage.slot).ok()?;
255                    if let Some(slot_info) = self.handle_struct(
256                        &storage.label,
257                        storage_type,
258                        slot_u256,
259                        struct_start_slot,
260                        storage.offset,
261                        &slot_str,
262                        0,
263                    ) {
264                        return Some(slot_info);
265                    }
266                }
267            } else if storage_type.encoding == ENCODING_MAPPING
268                && let Some(mapping_slots) = mapping_slots
269                && let Some(slot_info) =
270                    self.handle_mapping(storage, storage_type, slot, &slot_str, mapping_slots)
271            {
272                return Some(slot_info);
273            }
274        }
275
276        None
277    }

279    /// Handles identification of array slots.
280    ///
281    /// # Arguments
282    /// * `storage` - The storage metadata from the layout
283    /// * `storage_type` - Type information for the storage slot
284    /// * `slot` - The target slot being identified
285    /// * `array_start_slot` - The starting slot of the array in storage i.e base_slot
286    /// * `slot_str` - String representation of the slot for output
287    fn handle_array_slot(
288        &self,
289        storage: &Storage,
290        storage_type: &StorageType,
291        slot: U256,
292        array_start_slot: U256,
293        slot_str: &str,
294    ) -> Option<SlotInfo> {
295        // Check if slot is within array bounds
296        let total_bytes = storage_type.number_of_bytes.parse::<u64>().ok()?;
297        let total_slots = total_bytes.div_ceil(32);
298
299        if slot >= array_start_slot && slot < array_start_slot + U256::from(total_slots) {
300            let parsed_type = DynSolType::parse(&storage_type.label).ok()?;
301            let index = (slot - array_start_slot).to::<u64>();
302            // Format the array element label based on array dimensions
303            let label = match &parsed_type {
304                DynSolType::FixedArray(inner, _) => {
305                    if let DynSolType::FixedArray(_, inner_size) = inner.as_ref() {
306                        // 2D array: calculate row and column
307                        let row = index / (*inner_size as u64);
308                        let col = index % (*inner_size as u64);
309                        format!("{}[{row}][{col}]", storage.label)
310                    } else {
311                        // 1D array
312                        format!("{}[{index}]", storage.label)
313                    }
314                }
315                _ => storage.label.clone(),
316            };
317
318            return Some(SlotInfo {
319                label,
320                slot_type: StorageTypeInfo {
321                    label: storage_type.label.clone(),
322                    dyn_sol_type: parsed_type,
323                },
324                offset: 0,
325                slot: slot_str.to_string(),
326                members: None,
327                decoded: None,
328                keys: None,
329            });
330        }
331
332        None
333    }

335    /// Handles identification of struct slots.
336    ///
337    /// Recursively resolves struct members to find the exact member corresponding
338    /// to the target slot. Handles both single-slot (packed) and multi-slot structs.
339    ///
340    /// # Arguments
341    /// * `base_label` - The label/name for this struct or member
342    /// * `storage_type` - Type information for the storage
343    /// * `target_slot` - The target slot being identified
344    /// * `struct_start_slot` - The starting slot of this struct
345    /// * `offset` - Offset within the slot (for packed storage)
346    /// * `slot_str` - String representation of the slot for output
347    /// * `depth` - Current recursion depth
348    #[allow(clippy::too_many_arguments)]
349    fn handle_struct(
350        &self,
351        base_label: &str,
352        storage_type: &StorageType,
353        target_slot: U256,
354        struct_start_slot: U256,
355        offset: i64,
356        slot_str: &str,
357        depth: usize,
358    ) -> Option<SlotInfo> {
359        // Limit recursion depth to prevent stack overflow
360        const MAX_DEPTH: usize = 10;
361        if depth > MAX_DEPTH {
362            return None;
363        }
364
365        let members = storage_type
366            .other
367            .get("members")
368            .and_then(|v| serde_json::from_value::<Vec<Storage>>(v.clone()).ok())?;
369
370        // If this is the exact slot we're looking for (struct's base slot)
371        if struct_start_slot == target_slot
372        // Find the member at slot offset 0 (the member that starts at this slot)
373            && let Some(first_member) = members.iter().find(|m| m.slot == "0")
374        {
375            let member_type_info = self.storage_layout.types.get(&first_member.storage_type)?;
376
377            // Check if we have a single-slot struct (all members have slot "0")
378            let is_single_slot = members.iter().all(|m| m.slot == "0");
379
380            if is_single_slot {
381                // Build member info for single-slot struct
382                let mut member_infos = Vec::new();
383                for member in &members {
384                    if let Some(member_type_info) =
385                        self.storage_layout.types.get(&member.storage_type)
386                        && let Some(member_type) = DynSolType::parse(&member_type_info.label).ok()
387                    {
388                        member_infos.push(SlotInfo {
389                            label: member.label.clone(),
390                            slot_type: StorageTypeInfo {
391                                label: member_type_info.label.clone(),
392                                dyn_sol_type: member_type,
393                            },
394                            offset: member.offset,
395                            slot: slot_str.to_string(),
396                            members: None,
397                            decoded: None,
398                            keys: None,
399                        });
400                    }
401                }
402
403                // Build the CustomStruct type
404                let struct_name =
405                    storage_type.label.strip_prefix("struct ").unwrap_or(&storage_type.label);
406                let prop_names: Vec<String> = members.iter().map(|m| m.label.clone()).collect();
407                let member_types: Vec<DynSolType> =
408                    member_infos.iter().map(|info| info.slot_type.dyn_sol_type.clone()).collect();
409
410                let parsed_type = DynSolType::CustomStruct {
411                    name: struct_name.to_string(),
412                    prop_names,
413                    tuple: member_types,
414                };
415
416                return Some(SlotInfo {
417                    label: base_label.to_string(),
418                    slot_type: StorageTypeInfo {
419                        label: storage_type.label.clone(),
420                        dyn_sol_type: parsed_type,
421                    },
422                    offset,
423                    slot: slot_str.to_string(),
424                    decoded: None,
425                    members: if member_infos.is_empty() { None } else { Some(member_infos) },
426                    keys: None,
427                });
428            } else {
429                // Multi-slot struct - return the first member.
430                let member_label = format!("{}.{}", base_label, first_member.label);
431
432                // If the first member is itself a struct, recurse
433                if is_struct(&member_type_info.label) {
434                    return self.handle_struct(
435                        &member_label,
436                        member_type_info,
437                        target_slot,
438                        struct_start_slot,
439                        first_member.offset,
440                        slot_str,
441                        depth + 1,
442                    );
443                }
444
445                // Return the first member as a primitive
446                return Some(SlotInfo {
447                    label: member_label,
448                    slot_type: StorageTypeInfo {
449                        label: member_type_info.label.clone(),
450                        dyn_sol_type: DynSolType::parse(&member_type_info.label).ok()?,
451                    },
452                    offset: first_member.offset,
453                    slot: slot_str.to_string(),
454                    decoded: None,
455                    members: None,
456                    keys: None,
457                });
458            }
459        }
460
461        // Not the base slot - search through members
462        for member in &members {
463            let member_slot_offset = U256::from_str(&member.slot).ok()?;
464            let member_slot = struct_start_slot + member_slot_offset;
465            let member_type_info = self.storage_layout.types.get(&member.storage_type)?;
466            let member_label = format!("{}.{}", base_label, member.label);
467
468            // If this member is a struct, recurse into it
469            if is_struct(&member_type_info.label) {
470                let slot_info = self.handle_struct(
471                    &member_label,
472                    member_type_info,
473                    target_slot,
474                    member_slot,
475                    member.offset,
476                    slot_str,
477                    depth + 1,
478                );
479
480                if member_slot == target_slot || slot_info.is_some() {
481                    return slot_info;
482                }
483            }
484
485            if member_slot == target_slot {
486                // Found the exact member slot
487
488                // Regular member
489                let member_type = DynSolType::parse(&member_type_info.label).ok()?;
490                return Some(SlotInfo {
491                    label: member_label,
492                    slot_type: StorageTypeInfo {
493                        label: member_type_info.label.clone(),
494                        dyn_sol_type: member_type,
495                    },
496                    offset: member.offset,
497                    slot: slot_str.to_string(),
498                    members: None,
499                    decoded: None,
500                    keys: None,
501                });
502            }
503        }
504
505        None
506    }

508    /// Handles identification of mapping slots.
509    ///
510    /// Identifies mapping entries by walking up the parent chain to find the base slot,
511    /// then decodes the keys and builds the appropriate label.
512    ///
513    /// # Arguments
514    /// * `storage` - The storage metadata from the layout
515    /// * `storage_type` - Type information for the storage
516    /// * `slot` - The accessed slot being identified
517    /// * `slot_str` - String representation of the slot for output
518    /// * `mapping_slots` - Tracked mapping slot accesses for key resolution
519    fn handle_mapping(
520        &self,
521        storage: &Storage,
522        storage_type: &StorageType,
523        slot: &B256,
524        slot_str: &str,
525        mapping_slots: &MappingSlots,
526    ) -> Option<SlotInfo> {
527        trace!(
528            "handle_mapping: storage.slot={}, slot={:?}, has_keys={}, has_parents={}",
529            storage.slot,
530            slot,
531            mapping_slots.keys.contains_key(slot),
532            mapping_slots.parent_slots.contains_key(slot)
533        );
534
535        // Verify it's actually a mapping type
536        if storage_type.encoding != ENCODING_MAPPING {
537            return None;
538        }
539
540        // Check if this slot is a known mapping entry
541        if !mapping_slots.keys.contains_key(slot) {
542            return None;
543        }
544
545        // Convert storage.slot to B256 for comparison
546        let storage_slot_b256 = B256::from(U256::from_str(&storage.slot).ok()?);
547
548        // Walk up the parent chain to collect keys and validate the base slot
549        let mut current_slot = *slot;
550        let mut keys_to_decode = Vec::new();
551        let mut found_base = false;
552
553        while let Some((key, parent)) =
554            mapping_slots.keys.get(&current_slot).zip(mapping_slots.parent_slots.get(&current_slot))
555        {
556            keys_to_decode.push(*key);
557
558            // Check if the parent is our base storage slot
559            if *parent == storage_slot_b256 {
560                found_base = true;
561                break;
562            }
563
564            // Move up to the parent for the next iteration
565            current_slot = *parent;
566        }
567
568        if !found_base {
569            trace!("Mapping slot {} does not match any parent in chain", storage.slot);
570            return None;
571        }
572
573        // Resolve the mapping type to get all key types and the final value type
574        let (key_types, value_type_label, full_type_label) =
575            self.resolve_mapping_type(&storage.storage_type)?;
576
577        // Reverse keys to process from outermost to innermost
578        keys_to_decode.reverse();
579
580        // Build the label with decoded keys and collect decoded key values
581        let mut label = storage.label.clone();
582        let mut decoded_keys = Vec::new();
583
584        // Decode each key using the corresponding type
585        for (i, key) in keys_to_decode.iter().enumerate() {
586            if let Some(key_type_label) = key_types.get(i)
587                && let Ok(sol_type) = DynSolType::parse(key_type_label)
588                && let Ok(decoded) = sol_type.abi_decode(&key.0)
589            {
590                let decoded_key_str = format_token_raw(&decoded);
591                decoded_keys.push(decoded_key_str.clone());
592                label = format!("{label}[{decoded_key_str}]");
593            } else {
594                let hex_key = hex::encode_prefixed(key.0);
595                decoded_keys.push(hex_key.clone());
596                label = format!("{label}[{hex_key}]");
597            }
598        }
599
600        // Parse the final value type for decoding
601        let dyn_sol_type = DynSolType::parse(&value_type_label).unwrap_or(DynSolType::Bytes);
602
603        Some(SlotInfo {
604            label,
605            slot_type: StorageTypeInfo { label: full_type_label, dyn_sol_type },
606            offset: storage.offset,
607            slot: slot_str.to_string(),
608            members: None,
609            decoded: None,
610            keys: Some(decoded_keys),
611        })
612    }
613
614    fn resolve_mapping_type(&self, type_ref: &str) -> Option<(Vec<String>, String, String)> {
615        let storage_type = self.storage_layout.types.get(type_ref)?;
616
617        if storage_type.encoding != ENCODING_MAPPING {
618            // Not a mapping, return the type as-is
619            return Some((vec![], storage_type.label.clone(), storage_type.label.clone()));
620        }
621
622        // Get key and value type references
623        let key_type_ref = storage_type.key.as_ref()?;
624        let value_type_ref = storage_type.value.as_ref()?;
625
626        // Resolve the key type
627        let key_type = self.storage_layout.types.get(key_type_ref)?;
628        let mut key_types = vec![key_type.label.clone()];
629
630        // Check if the value is another mapping (nested case)
631        if let Some(value_storage_type) = self.storage_layout.types.get(value_type_ref) {
632            if value_storage_type.encoding == "mapping" {
633                // Recursively resolve the nested mapping
634                let (nested_keys, final_value, _) = self.resolve_mapping_type(value_type_ref)?;
635                key_types.extend(nested_keys);
636                return Some((key_types, final_value, storage_type.label.clone()));
637            } else {
638                // Value is not a mapping, we're done
639                return Some((
640                    key_types,
641                    value_storage_type.label.clone(),
642                    storage_type.label.clone(),
643                ));
644            }
645        }
646
647        None
648    }
649}

651/// Returns the base indices for array types, e.g. "\[0\]\[0\]" for 2D arrays.
652fn get_array_base_indices(dyn_type: &DynSolType) -> String {
653    match dyn_type {
654        DynSolType::FixedArray(inner, _) => {
655            if let DynSolType::FixedArray(_, _) = inner.as_ref() {
656                // Nested array (2D or higher)
657                format!("[0]{}", get_array_base_indices(inner))
658            } else {
659                // Simple 1D array
660                "[0]".to_string()
661            }
662        }
663        _ => String::new(),
664    }
665}
